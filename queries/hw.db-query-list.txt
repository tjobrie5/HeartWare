// create the database file with sqlite3
sqlite3 hw.db

// modify output
.mode columns
.headers on

// show tables
.tables

// creating a table called profiles 
CREATE TABLE profiles
    ( profileId INTEGER PRIMARY KEY, userName TEXT NOT NULL,
    sex TEXT, favExercise TEXT, disabilities TEXT,
    workoutLocation TEXT, workoutLevel TEXT, steps INTEGER );

CREATE TABLE meetups 
	( meetupsId INTEGER PRIMARY KEY,
		date TEXT,
		FOREIGN KEY (profileId) REFERENCES profiles(meetupsId) );
